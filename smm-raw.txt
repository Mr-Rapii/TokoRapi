<?php

namespace App\Models;

use Illuminate\Contracts\Auth\MustVerifyEmail;
use Illuminate\Database\Eloquent\Factories\HasFactory;
use Illuminate\Foundation\Auth\User as Authenticatable;
use Illuminate\Notifications\Notifiable;
use Laravel\Sanctum\HasApiTokens;

class User extends Authenticatable
{
    use HasApiTokens, HasFactory, Notifiable;

    protected $fillable = [
        'name', 'email', 'phone', 'google_id', 'password',
        'balance', 'is_admin', 'is_active', 'email_verified_at',
    ];

    protected $hidden = ['password', 'remember_token'];

    protected $casts = [
        'email_verified_at' => 'datetime',
        'password' => 'hashed',
        'balance' => 'decimal:2',
        'is_admin' => 'boolean',
        'is_active' => 'boolean',
    ];

    public function orders()
    {
        return $this->hasMany(Order::class);
    }

    public function transactions()
    {
        return $this->hasMany(Transaction::class);
    }

    public function topupTransactions()
    {
        return $this->hasMany(Transaction::class)->where('type', 'topup');
    }

    public function getFormattedBalanceAttribute(): string
    {
        return 'Rp ' . number_format($this->balance, 0, ',', '.');
    }

    public function addBalance(float $amount): void
    {
        $this->increment('balance', $amount);
    }

    public function deductBalance(float $amount): bool
    {
        if ($this->balance < $amount) {
            return false;
        }
        $this->decrement('balance', $amount);
        return true;
    }
}

    protected $fillable = [
        'service_id', 'name', 'type', 'category',
        'rate', 'custom_rate', 'min', 'max', 'is_active', 'description',
    ];

    protected $casts = [
        'rate' => 'decimal:4',
        'custom_rate' => 'decimal:4',
        'min' => 'integer',
        'max' => 'integer',
        'is_active' => 'boolean',
    ];

    public function orders()
    {
        return $this->hasMany(Order::class);
    }

    public function getEffectiveRateAttribute(): float
    {
        return $this->custom_rate ?? $this->rate;
    }

    public function calculatePrice(int $quantity): float
    {
        return ($quantity / 1000) * $this->effective_rate;
    }

    public function scopeActive($query)
    {
        return $query->where('is_active', true);
    }
}

<?php

namespace App\Http\Controllers\Auth;

use App\Http\Controllers\Controller;
use App\Models\User;
use App\Services\WhatsAppService;
use Illuminate\Support\Facades\Auth;
use Illuminate\Support\Facades\Log;
use Laravel\Socialite\Facades\Socialite;

class GoogleController extends Controller
{
    public function redirect()
    {
        return Socialite::driver('google')->redirect();
    }

    public function callback()
    {
        try {
            $googleUser = Socialite::driver('google')->user();

            $user = User::where('google_id', $googleUser->getId())
                ->orWhere('email', $googleUser->getEmail())
                ->first();

            if ($user) {
                if (!$user->google_id) {
                    $user->update(['google_id' => $googleUser->getId()]);
                }
            } else {
                $user = User::create([
                    'name' => $googleUser->getName(),
                    'email' => $googleUser->getEmail(),
                    'google_id' => $googleUser->getId(),
                    'email_verified_at' => now(),
                    'balance' => 0,
                    'is_admin' => false,
                    'is_active' => true,
                ]);

                try {
                    app(WhatsAppService::class)->sendWelcomeMessage($user);
                } catch (\Exception $e) {
                    Log::error('WA Welcome Error: ' . $e->getMessage());
                }
            }

            if (!$user->is_active) {
                return redirect()->route('login')->withErrors(['email' => 'Akun Anda dinonaktifkan.']);
            }

            Auth::login($user, true);
            return redirect()->intended(route('dashboard'));
        } catch (\Exception $e) {
            Log::error('Google OAuth Error: ' . $e->getMessage());
            return redirect()->route('login')->withErrors(['email' => 'Login Google gagal. Coba lagi.']);
        }
    }
}

<?php

namespace App\Http\Controllers\Auth;

use App\Http\Controllers\Controller;
use Illuminate\Auth\Events\PasswordReset;
use Illuminate\Http\Request;
use Illuminate\Support\Facades\Hash;
use Illuminate\Support\Facades\Password;
use Illuminate\Support\Str;

class PasswordResetController extends Controller
{
    public function showForgotForm()
    {
        return view('auth.forgot-password');
    }

    public function sendResetLink(Request $request)
    {
        $request->validate(['email' => 'required|email']);

        $status = Password::sendResetLink($request->only('email'));

        return $status === Password::RESET_LINK_SENT
            ? back()->with('success', __($status))
            : back()->withErrors(['email' => __($status)]);
    }

    public function showResetForm(Request $request, string $token)
    {
        return view('auth.reset-password', [
            'token' => $token,
            'email' => $request->email,
        ]);
    }

    public function reset(Request $request)
    {
        $request->validate([
            'token' => 'required',
            'email' => 'required|email',
            'password' => 'required|min:8|confirmed',
        ]);

        $status = Password::reset(
            $request->only('email', 'password', 'password_confirmation', 'token'),
            function (User $user, string $password) {
                $user->forceFill(['password' => Hash::make($password)])
                    ->setRememberToken(Str::random(60));
                $user->save();
                event(new PasswordReset($user));
            }
        );

        return $status === Password::PASSWORD_RESET
            ? redirect()->route('login')->with('success', 'Password berhasil direset.')
            : back()->withErrors(['email' => [__($status)]]);
    }
}

<?php

namespace App\Http\Controllers;

use App\Models\Order;
use App\Models\Service;
use App\Models\Transaction;
use App\Services\EzkifyService;
use App\Services\WhatsAppService;
use Illuminate\Http\Request;
use Illuminate\Support\Facades\Auth;
use Illuminate\Support\Facades\DB;
use Illuminate\Support\Facades\Log;

class OrderController extends Controller
{
    public function __construct(
        private EzkifyService $ezkify,
        private WhatsAppService $whatsApp
    ) {}

    public function index()
    {
        $categories = Service::active()->distinct()->pluck('category')->sort()->values();
        $services = Service::active()->get()->groupBy('category');
        return view('dashboard.order', compact('categories', 'services'));
    }

    public function getServices(Request $request)
    {
        $services = Service::active()
            ->when($request->category, fn($q) => $q->where('category', $request->category))
            ->get(['id', 'service_id', 'name', 'category', 'rate', 'custom_rate', 'min', 'max', 'description']);

        return response()->json($services->map(function ($s) {
            return [
                'id' => $s->id,
                'service_id' => $s->service_id,
                'name' => $s->name,
                'rate' => $s->effective_rate,
                'min' => $s->min,
                'max' => $s->max,
                'description' => $s->description,
            ];
        }));
    }

    public function store(Request $request)
    {
        $request->validate([
            'service_id' => 'required|exists:services,id',
            'link' => 'required|url',
            'quantity' => 'required|integer|min:1',
        ]);

        $service = Service::active()->findOrFail($request->service_id);
        $user = Auth::user();

        if ($request->quantity < $service->min || $request->quantity > $service->max) {
            return back()->withErrors(['quantity' => "Quantity harus antara {$service->min} - {$service->max}"]);
        }

        $price = $service->calculatePrice($request->quantity);

        if ($user->balance < $price) {
            return back()->withErrors(['balance' => 'Saldo tidak cukup. Silakan top up terlebih dahulu.']);
        }

        DB::beginTransaction();
        try {
            $user->deductBalance($price);

            $order = Order::create([
                'user_id' => $user->id,
                'service_id' => $service->id,
                'link' => $request->link,
                'quantity' => $request->quantity,
                'price' => $price,
                'status' => 'pending',
                'notes' => $request->notes,
            ]);

            Transaction::create([
                'user_id' => $user->id,
                'amount' => -$price,
                'type' => 'order_deduction',
                'reference' => 'ORDER-' . $order->id,
                'status' => 'approved',
                'notes' => "Order #{$order->id} - {$service->name}",
            ]);

            $apiResult = $this->ezkify->addOrder($service->service_id, $request->link, $request->quantity);

            if ($apiResult['success']) {
                $order->update([
                    'api_order_id' => $apiResult['order_id'],
                    'status' => 'processing',
                ]);
            } else {
                $order->update(['status' => 'failed', 'notes' => $apiResult['message'] ?? 'API Error']);
                $user->addBalance($price);
                Transaction::create([
                    'user_id' => $user->id,
                    'amount' => $price,
                    'type' => 'refund',
                    'reference' => 'REFUND-ORDER-' . $order->id,

<?php

namespace App\Services;

use App\Models\Service;
use Illuminate\Support\Facades\Http;
use Illuminate\Support\Facades\Log;

class EzkifyService
{
    private string $apiKey;
    private string $apiUrl;

    public function __construct()
    {
        $this->apiKey = config('services.ezkify.api_key');
        $this->apiUrl = config('services.ezkify.api_url', 'https://ezkify.com/api/v2');
    }

    private function request(array $params): array
    {
        try {
            $response = Http::timeout(30)->asForm()->post($this->apiUrl, array_merge([
                'key' => $this->apiKey,
            ], $params));

            if ($response->successful()) {
                $data = $response->json();
                return $data ?? [];
            }

            Log::error("Ezkify API Error: " . $response->status() . " - " . $response->body());
            return ['error' => 'API request failed'];
        } catch (\Exception $e) {
            Log::error("Ezkify Exception: " . $e->getMessage());
            return ['error' => $e->getMessage()];
        }
    }

    public function getServices(): array
    {
        return $this->request(['action' => 'services']);
    }

    public function syncServices(): array
    {
        $apiServices = $this->getServices();

        if (isset($apiServices['error'])) {
            throw new \Exception('Gagal mengambil services dari API: ' . $apiServices['error']);
        }

        $synced = 0;
        foreach ($apiServices as $apiService) {
            if (!isset($apiService['service'])) continue;

            Service::updateOrCreate(
                ['service_id' => (string) $apiService['service']],
                [
                    'name' => $apiService['name'] ?? 'Unknown',
                    'type' => $apiService['type'] ?? null,
                    'category' => $apiService['category'] ?? 'Lainnya',
                    'rate' => (float) ($apiService['rate'] ?? 0),
                    'min' => (int) ($apiService['min'] ?? 10),
                    'max' => (int) ($apiService['max'] ?? 10000),
                    'description' => $apiService['description'] ?? null,
                ]
            );
            $synced++;
        }

        return ['synced' => $synced];
    }

    public function addOrder(string $serviceId, string $link, int $quantity): array
    {
        $result = $this->request([
            'action' => 'add',
            'service' => $serviceId,
            'link' => $link,
            'quantity' => $quantity,
        ]);

        if (isset($result['order'])) {
            return ['success' => true, 'order_id' => (string) $result['order']];
        }

        return ['success' => false, 'message' => $result['error'] ?? 'Unknown error'];
    }

    public function getOrderStatus(string $orderId): array
    {
        $result = $this->request([
            'action' => 'status',
            'order' => $orderId,
        ]);

        if (isset($result['status'])) {
            return [
                'success' => true,
                'status' => strtolower($result['status']),
                'start_count' => $result['start_count'] ?? null,
                'remains' => $result['remains'] ?? null,
            ];
        }

        return ['success' => false, 'message' => $result['error'] ?? 'Unknown error'];
    }

    public function getMultipleOrderStatus(array $orderIds): array
    {
        $result = $this->request([
            'action' => 'status',
            'orders' => implode(',', $orderIds),
        ]);

        return $result;
    }

    public function getBalance(): string
    {
        $result = $this->request(['action' => 'balance']);
        return isset($result['balance']) ? 'USD ' . $result['balance'] : 'N/A';
    }

    public function cancelOrder(string $orderId): array
    {
        $result = $this->request([
            'action' => 'cancel',
            'orders' => $orderId,
        ]);

        return ['success' => !isset($result['error']), 'result' => $result];
    }
}

<?php

namespace App\Console\Commands;

use App\Models\Order;
use App\Services\EzkifyService;
use App\Services\WhatsAppService;
use Illuminate\Console\Command;
use Illuminate\Support\Facades\Log;

class CheckOrderStatus extends Command
{
    protected $signature = 'smm:check-orders';
    protected $description = 'Check dan update status orders dari Ezkify API';

    public function handle(EzkifyService $ezkify, WhatsAppService $whatsApp): int
    {
        $this->info('Checking order statuses...');

        $orders = Order::whereNotNull('api_order_id')
            ->whereNotIn('status', ['completed', 'cancelled', 'failed'])
            ->with('user')
            ->get();

        if ($orders->isEmpty()) {
            $this->info('Tidak ada order yang perlu dicek.');
            return Command::SUCCESS;
        }

        $this->info("Checking {$orders->count()} orders...");
        $updated = 0;

        foreach ($orders as $order) {
            try {
                $result = $ezkify->getOrderStatus($order->api_order_id);

                if (!$result['success']) continue;

                $newStatus = $this->mapStatus($result['status']);
                $wasCompleted = $order->status !== 'completed' && $newStatus === 'completed';

                $updateData = ['status' => $newStatus];
                if (isset($result['start_count'])) $updateData['start_count'] = $result['start_count'];
                if (isset($result['remains'])) $updateData['remains'] = $result['remains'];

                $order->update($updateData);
                $updated++;

                if ($wasCompleted && $order->user->phone) {
                    try {
                        $whatsApp->sendOrderCompleted($order->user, $order);
                    } catch (\Exception $e) {
                        Log::error('WA Completed Error: ' . $e->getMessage());
                    }
                }

                $this->line("Order #{$order->id}: {$order->status} → {$newStatus}");
            } catch (\Exception $e) {
                Log::error("CheckOrder Error #{$order->id}: " . $e->getMessage());
            }
        }

        $this->info("✅ Updated {$updated} orders.");
        return Command::SUCCESS;
    }

    private function mapStatus(string $status): string
    {
        return match (strtolower($status)) {
            'pending' => 'pending',
            'processing' => 'processing',
            'in progress', 'inprogress' => 'in_progress',
            'completed' => 'completed',
            'partial' => 'partial',
            'cancelled', 'canceled' => 'cancelled',
            'fail', 'failed' => 'failed',
            default => 'processing',

Route::middleware('guest')->group(function () {
    Route::get('/login', fn() => view('auth.login'))->name('login');
    Route::post('/login', function (\Illuminate\Http\Request $request) {
        $credentials = $request->validate([
            'email' => 'required|email',
            'password' => 'required',
        ]);

        if (Auth::attempt($credentials, $request->boolean('remember'))) {
            $request->session()->regenerate();
            if (Auth::user()->is_admin) {
                return redirect()->route('admin.dashboard');
            }
            return redirect()->intended(route('dashboard'));
        }

        return back()->withErrors(['email' => 'Email atau password salah.'])->onlyInput('email');
    })->name('login.submit');

    Route::get('/register', fn() => view('auth.register'))->name('register');
    Route::post('/register', function (\Illuminate\Http\Request $request) {
        $request->validate([
            'name' => 'required|string|max:255',
            'email' => 'required|email|unique:users',
            'phone' => 'nullable|string|max:20',
            'password' => 'required|min:8|confirmed',
        ]);

        $user = \App\Models\User::create([
            'name' => $request->name,
            'email' => $request->email,
            'phone' => $request->phone,
            'password' => bcrypt($request->password),
        ]);

        try {
            app(\App\Services\WhatsAppService::class)->sendWelcomeMessage($user);
        } catch (\Exception $e) {}

        Auth::login($user);
        return redirect()->route('dashboard');
    })->name('register.submit');

    Route::get('/forgot-password', [PasswordResetController::class, 'showForgotForm'])->name('password.request');
    Route::post('/forgot-password', [PasswordResetController::class, 'sendResetLink'])->name('password.email');

});

<script>
const fileUploadArea = document.getElementById('fileUploadArea');
const paymentProof = document.getElementById('paymentProof');
const fileUploadContent = document.getElementById('fileUploadContent');
const filePreview = document.getElementById('filePreview');
const previewImg = document.getElementById('previewImg');
const fileName = document.getElementById('fileName');

fileUploadArea.addEventListener('click', () => paymentProof.click());

fileUploadArea.addEventListener('dragover', (e) => {
    e.preventDefault();
    fileUploadArea.classList.add('drag-over');
});

fileUploadArea.addEventListener('dragleave', () => fileUploadArea.classList.remove('drag-over'));

fileUploadArea.addEventListener('drop', (e) => {
    e.preventDefault();
    fileUploadArea.classList.remove('drag-over');
    const file = e.dataTransfer.files[0];
    if (file) handleFile(file);
});

paymentProof.addEventListener('change', (e) => {
    const file = e.target.files[0];
    if (file) handleFile(file);
});

function handleFile(file) {
    if (file.size > 5 * 1024 * 1024) {
        alert('File terlalu besar. Max 5MB.');
        return;
    }
    const reader = new FileReader();
    reader.onload = (e) => {
        previewImg.src = e.target.result;
        fileName.textContent = file.name;
        fileUploadContent.classList.add('d-none');
        filePreview.classList.remove('d-none');
    };
    reader.readAsDataURL(file);

    const dataTransfer = new DataTransfer();
    dataTransfer.items.add(file);
    paymentProof.files = dataTransfer.files;
}
</script>
@endpush

@push('scripts')
<script>
const userBalance = {{ auth()->user()->balance }};
let currentService = null;

document.getElementById('categorySelect').addEventListener('change', function() {
    const category = this.value;
    const serviceSelect = document.getElementById('serviceSelect');
    const serviceInfo = document.getElementById('serviceInfo');
    const pricePreview = document.getElementById('pricePreview');
    const submitBtn = document.getElementById('submitBtn');

    serviceSelect.innerHTML = '<option value="">-- Pilih Service --</option>';
    serviceInfo.classList.add('d-none');
    pricePreview.classList.add('d-none');
    submitBtn.disabled = true;
    currentService = null;

    if (!category) {
        serviceSelect.disabled = true;
        return;
    }

    serviceSelect.disabled = true;
    serviceSelect.innerHTML = '<option value="">Loading...</option>';

    fetch(`/order/services?category=${encodeURIComponent(category)}`)
        .then(r => r.json())
        .then(services => {
            serviceSelect.innerHTML = '<option value="">-- Pilih Service --</option>';
            services.forEach(s => {
                const opt = document.createElement('option');
                opt.value = s.id;
                opt.textContent = `#${s.service_id} - ${s.name}`;
                opt.dataset.service = JSON.stringify(s);
                serviceSelect.appendChild(opt);
            });
            serviceSelect.disabled = false;
        })
        .catch(() => {
            serviceSelect.innerHTML = '<option value="">Error loading services</option>';
        });
});

document.getElementById('serviceSelect').addEventListener('change', function() {
    const opt = this.options[this.selectedIndex];
    const serviceInfo = document.getElementById('serviceInfo');
    const pricePreview = document.getElementById('pricePreview');
    const submitBtn = document.getElementById('submitBtn');

    if (!opt.dataset.service) {
        serviceInfo.classList.add('d-none');
        pricePreview.classList.add('d-none');
        submitBtn.disabled = true;
        currentService = null;
        return;
    }

    currentService = JSON.parse(opt.dataset.service);
    document.getElementById('infoRate').textContent = `Rp ${formatRupiah(currentService.rate * 1000)} / 1000`;
    document.getElementById('infoMinMax').textContent = `${formatNumber(currentService.min)} - ${formatNumber(currentService.max)}`;

    const descEl = document.getElementById('infoDesc');
    if (currentService.description) {
        descEl.textContent = currentService.description;
        descEl.classList.remove('d-none');
    } else {
        descEl.classList.add('d-none');
    }

    const quantityInput = document.getElementById('quantityInput');
    quantityInput.min = currentService.min;
    quantityInput.max = currentService.max;
    quantityInput.placeholder = currentService.min;
    document.getElementById('quantityHint').textContent = `Min: ${formatNumber(currentService.min)} | Max: ${formatNumber(currentService.max)}`;

    serviceInfo.classList.remove('d-none');
    pricePreview.classList.remove('d-none');
    submitBtn.disabled = false;
    updatePrice();
});

document.getElementById('quantityInput').addEventListener('input', updatePrice);

function updatePrice() {
    if (!currentService) return;
    const qty = parseInt(document.getElementById('quantityInput').value) || 0;
    const price = (qty / 1000) * currentService.rate;
    document.getElementById('estimatedPrice').textContent = `Rp ${formatRupiah(price)}`;

    const submitBtn = document.getElementById('submitBtn');
    if (price > userBalance) {
        submitBtn.classList.add('btn-danger');
        submitBtn.classList.remove('btn-accent');
        submitBtn.textContent = 'Saldo Tidak Cukup';
        submitBtn.disabled = true;
    } else {
        submitBtn.classList.remove('btn-danger');
        submitBtn.classList.add('btn-accent');
        submitBtn.innerHTML = '<i class="bi bi-cart-check me-2"></i>Buat Order';
        submitBtn.disabled = false;
    }
}

function formatRupiah(num) {
    return Math.round(num).toLocaleString('id-ID');
}

function formatNumber(num) {
    return num.toLocaleString('id-ID');
}
</script>
@endpush

// =============================================
//  RAPI DEV SMM PANEL - Custom JS
// =============================================

document.addEventListener('DOMContentLoaded', function () {
    // Sidebar Toggle (Mobile)
    const sidebarToggle = document.getElementById('sidebarToggle');
    const sidebar = document.getElementById('sidebar');

    if (sidebarToggle && sidebar) {
        sidebarToggle.addEventListener('click', function () {
            sidebar.classList.toggle('open');
        });

        document.addEventListener('click', function (e) {
            if (sidebar.classList.contains('open') &&
                !sidebar.contains(e.target) &&
                !sidebarToggle.contains(e.target)) {
                sidebar.classList.remove('open');
            }
        });
    }

    // Auto-dismiss alerts
    const alerts = document.querySelectorAll('.alert.alert-dismissible');
    alerts.forEach(function (alert) {
        setTimeout(function () {
            const bsAlert = bootstrap.Alert.getOrCreateInstance(alert);
            if (bsAlert) bsAlert.close();
        }, 5000);
    });

    // Form loading states
    const forms = document.querySelectorAll('form');
    forms.forEach(function (form) {
        form.addEventListener('submit', function () {
            const submitBtn = form.querySelector('button[type="submit"]:not(.no-loading)');
            if (submitBtn && !submitBtn.dataset.noLoading) {
                const original = submitBtn.innerHTML;
                submitBtn.disabled = true;
                submitBtn.innerHTML = '<span class="spinner-border spinner-border-sm me-2"></span>Memproses...';
                setTimeout(function () {
                    submitBtn.disabled = false;
                    submitBtn.innerHTML = original;
                }, 8000);
            }
        });
    });

    // Tooltip init
    const tooltipTriggers = document.querySelectorAll('[data-bs-toggle="tooltip"]');
    tooltipTriggers.forEach(function (el) {
        new bootstrap.Tooltip(el);
    });

    // Copy to clipboard
    document.querySelectorAll('[data-copy]').forEach(function (el) {
        el.addEventListener('click', function () {
            const text = el.getAttribute('data-copy');
            navigator.clipboard.writeText(text).then(function () {
                const original = el.innerHTML;
                el.innerHTML = '<i class="bi bi-check-lg"></i>';
                setTimeout(function () { el.innerHTML = original; }, 1500);
            });
        });
    });

    // Number formatting in inputs
    const numberInputs = document.querySelectorAll('input[type="number"]');
    numberInputs.forEach(function (input) {
        input.addEventListener('wheel', function (e) {
            e.preventDefault();
        });
    });

    // Service rate modal fix
    document.querySelectorAll('[id^="rateModal"]').forEach(function (modal) {
        modal.addEventListener('show.bs.modal', function () {
            const input = modal.querySelector('input[name="custom_rate"]');
            if (input && input.value) {
                // value is per 1000, display as per 1000
            }
        });
    });
});
